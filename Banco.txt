create database bibellus;
use bibellus;

create table empresa (
nome varchar (200),
cnpj varchar (18),
inscricao varchar (15),
cnae varchar (7),
telefone varchar (13),
celular varchar (14),
email varchar (200),
site varchar (200),
endereco varchar (200),
cep varchar (10)
);

create table socios (
id int not null auto_increment primary key,
nome varchar (200),
cpf varchar (14)
);

drop table socios;

create table fornecedores (
id int not null primary key auto_increment,
empresa varchar (200),
categoria varchar (200)
);

drop table fornecedores;

create table despesas (
tipos varchar (200),
valor varchar (200),
observacao varchar (200)
) ;

create table vendas (
data date,
pagamento varchar (200),
valor varchar (200)
) ;

create table acompanhamento (
id int not null primary key auto_increment,
descricao varchar (200)
) ;

drop table acompanhamento;

create table sorvete (
id int not null primary key auto_increment,
sabor varchar (200)
) ;


create table descartavel (
id int not null primary key auto_increment,
tipo varchar (200)
) ;


create table unidade (
id int not null primary key auto_increment,
descricao varchar (200)
) ;

create table tipo (
id int not null primary key auto_increment,
descricao varchar (200)
) ;

create table produto (
id int not null primary key auto_increment,
id_tipo_fk int not null,
descricao varchar (200),
foreign key (id_tipo_fk) references tipo(id)
) ;

drop table produto;

create table estoque (
id int not null primary key auto_increment,
id_produto_fk int not null,
id_unidade_fk int not null,
quantidade int not null,
atualizacao date,
foreign key (id_produto_fk) references produto(id),
foreign key (id_unidade_fk) references unidade(id)
) ;

drop table estoque;
